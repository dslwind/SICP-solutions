(define (max x y z)
   (cond ((and (> x y) (> x z)) x)
         ((and (> y x) (> y z)) y)
         (else z)))

    (max (+ x y) (+ x z) (+ y z)))

(define (cube-root x)
    (cube-root-iter 1.0 x))

(define (cube-root-iter y x)
    (if (good-enough y x)
        y
        (cube-root-iter (improve y x) x)))                

(define (good-enough y x)
    (< (abs (- (cube y) x)) 
       0.0001))

(define (cube x)
    (* x x x))

(define (improve y x)
    (/ (+ (/ x (* y y))
          (* 2 y))
       3))

(define (abs x)
    (if (< x 0)
        (- x)
